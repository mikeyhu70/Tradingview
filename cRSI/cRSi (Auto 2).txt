crsi = 0.0
domcycle =  mode_domcycle == 'Auto' ?  math.max(auto_domcycle,math.round(lookback)) : _domcycle


cyclelen = domcycle / divider

//h1 = hline(30, color=color.silver, linestyle=hline.style_dashed)
//h2 = hline(70, color=color.silver, linestyle=hline.style_dashed)

auto_vibration = domcycle * 0.5 / 3 
vibration = mode_vibration == 'Auto' ? mode_domcycle == 'Auto' ? math.max(auto_vibration,math.round(1/factor*lookback)) : auto_vibration  : _vibration


auto_leveling = domcycle * 0.5 / 4 
leveling = mode_leveling == 'Auto' ?  mode_domcycle == 'Auto'  ? math.max(auto_leveling,math.round(1/factor*lookback)) : auto_leveling : _leveling



torque = 2.0 / (vibration + 1)
phasingLag = (vibration - 1) / 2.0


cyclicmemory = math.max(domcycle * 2,math.round(factor*lookback))
//set min/max ranges?


//up = ta.rma(math.max(ta.change(src), 0), cyclelen)
//down = ta.rma(-math.min(ta.change(src), 0), cyclelen)

upsrc = math.max(ta.change(src), 0)
downsrc = -math.min(ta.change(src), 0)
up = 0.0
down = 0.0
upsum = 0.0
downsum = 0.0

cyclelen1 = cyclelen

alpha = 1 / (cyclelen1)

for i = 0 to cyclelen1 - 1
    upsum := upsum + upsrc[i] / cyclelen1
up := na(up[1]) ? upsum : alpha * upsrc + (1 - alpha) * nz(up[1])


for i = 0 to cyclelen1 - 1
    downsum := downsum + downsrc[i] / cyclelen1
down := na(down[1]) ? downsum : alpha * downsrc + (1 - alpha) * nz(down[1])



rsi = down == 0 ? 100 : up == 0 ? 0 : 100 - 100 / (1 + up / down)
crsi := torque * (2 * rsi - rsi[phasingLag]) + (1 - torque) * nz(crsi[1])



lmax = -999999.0
lmin = 999999.0
for i = 0 to cyclicmemory - 1 by 1
    if nz(crsi[i], -999999.0) > lmax
        lmax := nz(crsi[i])
        lmax
    else
        if nz(crsi[i], 999999.0) < lmin
            lmin := nz(crsi[i])
            lmin

mstep = (lmax - lmin) / 100
aperc = leveling / 100

db = 0.0
for steps = 0 to 100 by 1
    testvalue = lmin + mstep * steps
    above = 0
    below = 0
    for m = 0 to cyclicmemory - 1 by 1
        below += (crsi[m] < testvalue ? 1 : 0)
        below

    ratio = below / cyclicmemory
    if ratio >= aperc
        db := testvalue
        break
    else
        continue

ub = 0.0
for steps = 0 to 100 by 1
    testvalue = lmax - mstep * steps
    above = 0
    for m = 0 to cyclicmemory - 1 by 1
        above += (crsi[m] >= testvalue ? 1 : 0)
        above

    ratio = above / cyclicmemory
    if ratio >= aperc
        ub := testvalue
        break
    else
        continue


lowband = plot(db, 'LowBand', color.new(color.aqua, 0),linewidth=2)
highband = plot(ub, 'HighBand', color.new(color.aqua, 0),linewidth=2)
midband = plot((db+ub)/2, 'MidBand', color.new(color.silver, 60),linewidth=1)
