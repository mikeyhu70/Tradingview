// ─── UNIVERSAL LOG/LIN PROJECTION ENGINE ───
calcProj(baseP, refA, refB, mult, dir, isLog) =>
    float safeBase = math.max(baseP, 1e-10)
    float safeA = math.max(refA, 1e-10)
    float safeB = math.max(refB, 1e-10)
    if isLog
        math.exp(math.log(safeBase) + dir * (math.abs(math.log(safeA) - math.log(safeB)) * mult))
    else
        baseP + dir * (math.abs(refA - refB) * mult)

// ─── Helper Functions ───
find_extremum(float[] px_arr, int start_idx, int end_idx, bool seek_max) =>
    float best_v = array.get(px_arr, start_idx)
    int best_i = start_idx
    if end_idx >= start_idx
        for i = start_idx to end_idx
            float v = array.get(px_arr, i)
            if seek_max
                if v > best_v
                    best_v := v
                    best_i := i
            else
                if v < best_v
                    best_v := v
                    best_i := i
    best_i

// Parity-aware: only considers indices where (i - origin) % 2 == parity
find_extremum_swing(float[] px_arr, int start_idx, int end_idx, bool seek_max, int origin_idx, int parity) =>
    float best_v = na
    int best_i = start_idx
    if end_idx >= start_idx
        for i = start_idx to end_idx
            if (i - origin_idx) % 2 == parity
                float v = array.get(px_arr, i)
                if na(best_v)
                    best_v := v
                    best_i := i
                else if seek_max and v > best_v
                    best_v := v
                    best_i := i
                else if not seek_max and v < best_v
                    best_v := v
                    best_i := i
    best_i

findStructuralWaves(isBull, px_arr, lookback) =>
    sz = array.size(px_arr)
    recentStart = math.max(0, sz - lookback) 
    int oIdx = find_extremum(px_arr, recentStart, sz - 1, not isBull) 
    int gExtI = find_extremum(px_arr, oIdx, sz - 1, isBull) 
    
    int w1I = oIdx, int w2I = oIdx, int w3I = oIdx, int w4I = oIdx, int w5I = oIdx
    int dist = gExtI - oIdx
    
    if dist < 3
        w1I := gExtI
        w2I := math.min(w1I + 1, sz - 1)
        w3I := math.min(w2I + 1, sz - 1)
        w4I := math.min(w3I + 1, sz - 1)
        w5I := math.min(w4I + 1, sz - 1)
    else if dist < 5
        w3I := gExtI
        w1I := find_extremum_swing(px_arr, oIdx + 1, w3I - 1, isBull, oIdx, 1)
        w2I := find_extremum_swing(px_arr, w1I + 1, w3I - 1, not isBull, oIdx, 0)
        w4I := math.min(w3I + 1, sz - 1)
        w5I := math.min(w4I + 1, sz - 1)
    else
        w5I := gExtI
        
        float bestScore = -1.0
        int bW1 = oIdx + 1, int bW2 = oIdx + 2, int bW3 = oIdx + 3, int bW4 = oIdx + 4
        
        float p0 = array.get(px_arr, oIdx)
        float p5 = array.get(px_arr, w5I)
        
        for i = oIdx + 1 to w5I - 4
            for j = i + 1 to w5I - 3
                for k = j + 1 to w5I - 2
                    for m = k + 1 to w5I - 1
                        float p1 = array.get(px_arr, i)
                        float p2 = array.get(px_arr, j)
                        float p3 = array.get(px_arr, k)
                        float p4 = array.get(px_arr, m)
                        
                        bool isValid = true

                        // ── Enforce swing type: zigzag alternates H-L-H-L ──
                        // W1,W3 must be opposite type from origin (odd offset)
                        // W2,W4 must be same type as origin (even offset)
                        if (i - oIdx) % 2 != 1
                            isValid := false
                        if (j - oIdx) % 2 != 0
                            isValid := false
                        if (k - oIdx) % 2 != 1
                            isValid := false
                        if (m - oIdx) % 2 != 0
                            isValid := false

                        if isValid and isBull
                            if p2 <= p0 or p3 <= p1 or p5 <= p3 or p4 <= p1
                                isValid := false
                        if isValid and not isBull
                            if p2 >= p0 or p3 >= p1 or p5 >= p3 or p4 >= p1
                                isValid := false
                                
                        if isValid
                            float len1 = math.abs(p1 - p0)
                            float len3 = math.abs(p3 - p2)
                            float len5 = math.abs(p5 - p4)
                            
                            if len3 < len1 and len3 < len5
                                isValid := false
                                
                            if isValid
                                float score = len3
                                if score > bestScore
                                    bestScore := score
                                    bW1 := i
                                    bW2 := j
                                    bW3 := k
                                    bW4 := m
        
        if bestScore > -1.0
            w1I := bW1
            w2I := bW2
            w3I := bW3
            w4I := bW4
        else
            int limit1 = oIdx + math.max(1, math.round(dist * 0.33))
            int limit2 = oIdx + math.max(2, math.round(dist * 0.66))
            
            w1I := find_extremum_swing(px_arr, oIdx + 1, limit1, isBull, oIdx, 1)
            w3I := find_extremum_swing(px_arr, math.max(w1I + 2, limit1 + 1), math.min(limit2, w5I - 2), isBull, oIdx, 1)
            w2I := find_extremum_swing(px_arr, w1I + 1, w3I - 1, not isBull, oIdx, 0)
            w4I := find_extremum_swing(px_arr, w3I + 1, w5I - 1, not isBull, oIdx, 0)
        
    [oIdx, w1I, w2I, w3I, w4I, w5I]

// ─── EW SCORING FUNCTION ───
scoreEWStructure(px_arr, pt_arr, sOrigin, sW1, sW2, sW3, sW4, sW5, isBull) =>
    float score = 0.0
    int sz = array.size(px_arr)
    
    if sz < 5 or sW5 >= sz or sOrigin < 0
        score := -1000.0
    else
        float p0 = array.get(px_arr, sOrigin)
        float p1 = sW1 > sOrigin and sW1 < sz ? array.get(px_arr, sW1) : p0
        float p2 = sW2 > sW1 and sW2 < sz ? array.get(px_arr, sW2) : p0
        float p3 = sW3 > sW2 and sW3 < sz ? array.get(px_arr, sW3) : p0
        float p4 = sW4 > sW3 and sW4 < sz ? array.get(px_arr, sW4) : p0
        float p5 = sW5 > sW4 and sW5 < sz ? array.get(px_arr, sW5) : p0
        
        float len1 = math.abs(p1 - p0)
        float len2 = math.abs(p2 - p1)
        float len3 = math.abs(p3 - p2)
        float len4 = math.abs(p4 - p3)
        float len5 = math.abs(p5 - p4)
        
        bool rule1 = true
        if isBull
            rule1 := p2 > p0
        else
            rule1 := p2 < p0
        if rule1
            score += 100.0
        else
            score -= 200.0
        
        bool rule2 = not (len3 < len1 and len3 < len5)
        if rule2
            score += 100.0
        else
            score -= 200.0
        
        bool rule3 = true
        if isBull
            rule3 := p4 > p1
        else
            rule3 := p4 < p1
        if rule3
            score += 100.0
        else
            score -= 200.0
        
        float w2Ret = len1 > 0 ? len2 / len1 : 0.0
        float w2Score = 0.0
        if w2Ret >= 0.382 and w2Ret <= 0.786
            w2Score := 50.0 - math.abs(w2Ret - 0.618) * 100.0
        else if w2Ret > 0.786 and w2Ret < 1.0
            w2Score := 10.0
        else
            w2Score := -20.0
        score += w2Score
        
        float w3Ext = len1 > 0 ? len3 / len1 : 0.0
        float w3Score = 0.0
        if w3Ext >= 1.618 and w3Ext <= 2.0
            w3Score := 80.0
        else if w3Ext >= 2.0 and w3Ext <= 2.618
            w3Score := 70.0
        else if w3Ext >= 2.618 and w3Ext <= 3.618
            w3Score := 60.0
        else if w3Ext >= 1.0 and w3Ext < 1.618
            w3Score := 30.0
        else if w3Ext > 3.618
            w3Score := 40.0
        else
            w3Score := -30.0
        score += w3Score
        
        float w4Ret = len3 > 0 ? len4 / len3 : 0.0
        float w4Score = 0.0
        if w4Ret >= 0.236 and w4Ret <= 0.5
            w4Score := 50.0 - math.abs(w4Ret - 0.382) * 80.0
        else if w4Ret > 0.5 and w4Ret <= 0.618
            w4Score := 20.0
        else if w4Ret > 0.618
            w4Score := -20.0
        else
            w4Score := 10.0
        score += w4Score
        
        float w5Ratio = len1 > 0 ? len5 / len1 : 0.0
        float w5Score = 0.0
        if w5Ratio >= 0.618 and w5Ratio <= 1.0
            w5Score := 40.0
        else if w5Ratio > 1.0 and w5Ratio <= 1.618
            w5Score := 50.0
        else if w5Ratio > 1.618 and w5Ratio <= 2.618
            w5Score := 30.0
        else if w5Ratio < 0.618 and w5Ratio > 0.382
            w5Score := 20.0
        else
            w5Score := -10.0
        score += w5Score
        
        bool alternation = (w2Ret > 0.5 and w4Ret < 0.382) or (w2Ret < 0.5 and w4Ret > 0.382)
        if alternation
            score += 30.0
        
        if len3 > len1 and len3 > len5
            score += 40.0
        
        int swingCount = 0
        if sW1 > sOrigin
            swingCount += 1
        if sW2 > sW1
            swingCount += 1
        if sW3 > sW2
            swingCount += 1
        if sW4 > sW3
            swingCount += 1
        if sW5 > sW4
            swingCount += 1
        score += swingCount * 10.0
        
    score
