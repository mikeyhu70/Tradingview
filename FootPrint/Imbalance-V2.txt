//@version=6
indicator("Footprint Imbalance Delta", overlay = true, max_labels_count = 500)

// --- Inputs --------------------------------------------------------------
ticksPerRow      = input.int(1, "Ticks per footprint row", minval = 1)
valueAreaPct     = input.int(70, "Value Area %", minval = 1, maxval = 100)
barsToShow       = input.int(1, "Show labels on last N bars", minval = 1, maxval = 500)

// Imbalance Settings
imbalanceFactor  = input.float(1.5, "Imbalance multiplier", minval = 1.0, tooltip = "How many times larger one side must be to count as an imbalance (e.g. 1.5 = 150%)")
imbMethod        = input.string("Diagonal Primary", "Imbalance Calculation Priority", options = ["Diagonal Primary", "Across (Horizontal) Primary", "Diagonal Only", "Across Only"])
minAbsImbalance  = input.float(50, "Min absolute imbalance delta to show", minval = 0)
showZeros        = input.bool(false, "Show zero imbalance delta")

// Display & Styling
textSizeInput    = input.string("normal", "Text size", options = ["tiny", "small", "normal", "large", "huge"])
deltaPosColor    = input.color(color.lime,  "Buy color")
deltaNegColor    = input.color(color.rgb(233, 108, 162),   "Sell color")
deltaZeroColor   = input.color(color.gray,  "Delta neutral color")
swingInProgressColor = input.color(color.white, "SI~ color")
displayMode      = input.string("Delta only", "Display mode", options = ["All", "Delta only"])

// ATR-based displacement (more responsive to volatility)
atrMultiplier    = input.float(1.5, "ATR displacement multiplier", minval = 0.1)
lineGapFactor    = input.float(0.20, "Label line gap (ATR fraction)", minval = 0.001, step = 0.01)
useATR           = input.bool(true, "Use ATR for spacing (vs fixed ticks)")
offsetTicks      = input.int(3, "Tick spacing (when ATR off)", minval = 1)
atrLength        = input.int(14, "ATR length", minval = 1)

// Swing imbalance
swingLen             = input.int(4, "Swing length", minval = 1)
showSwing            = input.bool(true, "Show swing delta")
swingsToShow         = input.int(40, "Number of swings to show", minval = 1)
swingAtrMultiplier   = input.float(0.1, "Swing ATR displacement multiplier", minval = 0.1)
siAtrMultiplier      = input.float(3.0, "SI~ ATR displacement multiplier", minval = 0.1)
siBarOffset         = input.int(30, "SI~ label bar offset", minval = 0)
siTextSize          = input.string("normal", "SI~ text size", options = ["tiny", "small", "normal", "large", "huge"])

// --- Initialization ------------------------------------------------------
footprint fp = request.footprint(ticksPerRow, valueAreaPct)
float atrMa  = ta.atr(atrLength)
float siAtr2 = ta.atr(2)

fmt(val) =>
    math.abs(val) >= 1000000 ? str.tostring(val / 1000000.0, "#.0") + "M" : 
     math.abs(val) >= 1000 ? str.tostring(val / 1000.0, "#.0") + "K" : 
     str.tostring(val)

// --- Imbalance math ------------------------------------------------------
float barBuyImb  = na
float barSellImb = na

if not na(fp)
    barBuyImb  := 0.0
    barSellImb := 0.0

    array<volume_row> rows = fp.rows()
    rowCount = array.size(rows)

    if rowCount > 0
        for i = 0 to rowCount - 1
            volume_row row = array.get(rows, i)
            float askVol   = row.buy_volume()
            float bidVol   = row.sell_volume()

            // Pre-calculate diagonal references
            bool hasLower = i > 0
            bool hasUpper = i < rowCount - 1
            
            float lowerBid = hasLower ? array.get(rows, i - 1).sell_volume() : 0.0
            float upperAsk = hasUpper ? array.get(rows, i + 1).buy_volume()  : 0.0

            // Determine if thresholds are met
            bool diagBuyTrigger   = hasLower and lowerBid > 0 and askVol >= lowerBid * imbalanceFactor
            bool diagSellTrigger  = hasUpper and upperAsk > 0 and bidVol >= upperAsk * imbalanceFactor
            
            bool acrossBuyTrigger = askVol >= bidVol * imbalanceFactor
            bool acrossSellTrigger= bidVol >= askVol * imbalanceFactor

            // Execution logic based on user priority selection
            bool buyCounted  = false
            bool sellCounted = false

            if imbMethod == "Diagonal Primary"
                // Check Diagonal First
                if diagBuyTrigger
                    barBuyImb += askVol - lowerBid
                    buyCounted := true
                // If no diagonal, check Across
                if not buyCounted and acrossBuyTrigger
                    barBuyImb += askVol - bidVol
                    
                // Check Diagonal First
                if diagSellTrigger
                    barSellImb += bidVol - upperAsk
                    sellCounted := true
                // If no diagonal, check Across
                if not sellCounted and acrossSellTrigger
                    barSellImb += bidVol - askVol

            else if imbMethod == "Across (Horizontal) Primary"
                if acrossBuyTrigger
                    barBuyImb += askVol - bidVol
                    buyCounted := true
                if not buyCounted and diagBuyTrigger
                    barBuyImb += askVol - lowerBid
                    
                if acrossSellTrigger
                    barSellImb += bidVol - askVol
                    sellCounted := true
                if not sellCounted and diagSellTrigger
                    barSellImb += bidVol - upperAsk

            else if imbMethod == "Diagonal Only"
                if diagBuyTrigger
                    barBuyImb += askVol - lowerBid
                if diagSellTrigger
                    barSellImb += bidVol - upperAsk

            else if imbMethod == "Across Only"
                if acrossBuyTrigger
                    barBuyImb += askVol - bidVol
                if acrossSellTrigger
                    barSellImb += bidVol - askVol

// Imbalance delta for the bar
float netImbalanceDelta = na
if not na(barBuyImb) and not na(barSellImb)
    netImbalanceDelta := barBuyImb - barSellImb


// --- Swing Logic (Fixed Lookback Lag) ------------------------------------
float ph = ta.pivothigh(high, swingLen, swingLen)
float pl = ta.pivotlow(low,  swingLen, swingLen)

var float swingAccum = 0.0
var int   swingNum   = 0
var float lastBarDelta = 0.0

if not na(netImbalanceDelta)
    swingAccum += netImbalanceDelta
    lastBarDelta := netImbalanceDelta

// Get the sum of the delta for the bars that occurred AFTER the actual pivot
float recentImbSum = math.sum(nz(netImbalanceDelta), swingLen)

float swingSpacing = useATR ? atrMa * swingAtrMultiplier : offsetTicks * syminfo.mintick
float siSpacing    = useATR ? atrMa * siAtrMultiplier : offsetTicks * syminfo.mintick
var label[] swingLabels = array.new<label>()

if showSwing and barstate.isconfirmed
    bool isPivot = not na(ph) or not na(pl)

    if isPivot
        swingNum += 1
        // Remove the recent bars from the completed swing
        float completedSwingVol = swingAccum - recentImbSum

        float pivotPrice = not na(ph) ? ph + swingSpacing : pl - swingSpacing
        color swingColor = completedSwingVol > 0 ? deltaPosColor : completedSwingVol < 0 ? deltaNegColor : deltaZeroColor

        // Draw the historical label AT the pivot bar
        array.push(swingLabels, label.new(bar_index - swingLen, pivotPrice,
             "SI:" + str.tostring(fmt(int(math.ceil(completedSwingVol)))),
             xloc = xloc.bar_index, yloc = yloc.price,
             style = label.style_none, textcolor = swingColor, size = size.normal))

        // Initialize the NEW in-progress swing with the volume that happened after the pivot
        swingAccum := recentImbSum

while array.size(swingLabels) > swingsToShow
    label.delete(array.shift(swingLabels))


// --- Formatting & Standard Labels ----------------------------------------
recentEnough  = bar_index >= math.max(last_bar_index - barsToShow + 1, 0)
passZero      = showZeros ? true : netImbalanceDelta != 0
passMin       = math.abs(nz(netImbalanceDelta)) >= minAbsImbalance
showLabel     = barstate.isconfirmed and not na(netImbalanceDelta) and recentEnough and passZero and passMin and not showSwing
showDeltaOnly = displayMode == "Delta only"

labelSize =
     textSizeInput == "tiny"   ? size.tiny  :
     textSizeInput == "small"  ? size.small :
     textSizeInput == "normal" ? size.normal :
     textSizeInput == "large"  ? size.large : size.huge



ipLabelSize = siTextSize == "tiny"   ? size.tiny  :
              siTextSize == "small"  ? size.small :
              siTextSize == "normal" ? size.normal :
              siTextSize == "large"  ? size.large : size.huge

int sellVal  = int(math.ceil(nz(barSellImb)))
int buyVal   = int(math.ceil(nz(barBuyImb)))
int deltaVal = int(math.ceil(nz(netImbalanceDelta)))

float spacing = useATR ? atrMa * atrMultiplier : offsetTicks * syminfo.mintick
float lineGap = atrMa * lineGapFactor

var label[] labels = array.new<label>()

if showLabel
    string sellTxt  = "S:" + fmt(sellVal)
    string buyTxt   = "B:" + fmt(buyVal)
    string deltaTxt = "I:" + fmt(deltaVal)
    color  deltaColor = deltaVal > 0 ? deltaPosColor : deltaVal < 0 ? deltaNegColor : deltaZeroColor

    float labelY = deltaVal < 0 ? high + spacing : low - spacing

    if showDeltaOnly
        array.push(labels, label.new(bar_index, labelY, deltaTxt,
             xloc = xloc.bar_index, yloc = yloc.price,
             style = label.style_none, textcolor = deltaColor, size = labelSize))
    else
        if deltaVal < 0
            array.push(labels, label.new(bar_index, labelY + lineGap * 2, buyTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaPosColor, size = labelSize))
            array.push(labels, label.new(bar_index, labelY + lineGap, sellTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaNegColor, size = labelSize))
            array.push(labels, label.new(bar_index, labelY, deltaTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaColor, size = labelSize))
        else
            array.push(labels, label.new(bar_index, labelY + lineGap * 2, deltaTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaColor, size = labelSize))
            array.push(labels, label.new(bar_index, labelY + lineGap, buyTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaPosColor, size = labelSize))
            array.push(labels, label.new(bar_index, labelY, sellTxt, xloc = xloc.bar_index, yloc = yloc.price, style = label.style_none, textcolor = deltaNegColor, size = labelSize))

int labelsPerBar = showDeltaOnly ? 1 : 3
while array.size(labels) > barsToShow * labelsPerBar + 20
    label.delete(array.shift(labels))

// --- In-progress Swing Label ---------------------------------------------
var label swingInProgressLabel = na

if showSwing
    float ipDelta = swingAccum
    string ipTxt  = "SI~:" + str.tostring(swingNum) + "|" + fmt(int(math.ceil(lastBarDelta)))
    float  ipY    = (ipDelta < 0 ? high + siSpacing : low - siSpacing) - siAtr2

    if na(swingInProgressLabel)
        swingInProgressLabel := label.new(bar_index + siBarOffset, ipY, ipTxt,
             xloc = xloc.bar_index, yloc = yloc.price,
             style = label.style_none, textcolor = swingInProgressColor, size = ipLabelSize)
    else
        label.set_x(swingInProgressLabel, bar_index + siBarOffset)
        label.set_y(swingInProgressLabel, ipY)
        label.set_text(swingInProgressLabel, ipTxt)
        label.set_textcolor(swingInProgressLabel, swingInProgressColor)
